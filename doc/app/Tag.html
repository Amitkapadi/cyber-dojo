def diff(m)
  command = "--ignore-space-at-eol --find-copies-harder #{@n} #{m} sandbox"
  diff_lines = clean(git.diff(@avatar.path, command))
  visible_files = @avatar.tags[m].visible_files
  git_diff(diff_lines, visible_files)
end
