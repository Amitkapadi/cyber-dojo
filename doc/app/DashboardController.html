def progress
  animals = { }
  avatars.active.each do |avatar|
    animals[avatar.name] = {
      :colour => avatar.lights[-1].colour,
      :progress => most_recent_progress(avatar)
    }
  end
  render :json => {
    :animals => animals
  }
end
